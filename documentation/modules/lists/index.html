Behavior

//  ====================================================
//  List defaults
//  ====================================================

ol {
	&.#{$list-base-class} {
		@extend .#{$list-base-class}-#{$list-ol-default};
	}
}
ul {
	&.#{$list-base-class} {
		@extend .#{$list-base-class}-#{$list-ul-default};
	}
}
